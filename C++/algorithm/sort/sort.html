# Quick Sort

  • 从数列中挑出一个元素，称为 "基准"（pivot）;
  • 重新排序数列，所有元素比基准值小的摆放在基准前面，所有元素比基准值大的摆在基准的后面（相同的数可以到任一边）。在这个分区退出之后，该基准就处于数列的中间位置。这个称为分区（partition）操作；
  • 递归地（recursive）把小于基准值元素的子数列和大于基准值元素的子数列排序；
void qsort(vector<int>& nums, int start, int end){
    if (start >= end) return ;
    int pivot = start, l = start, r = end;
    while(true){
        while(start <= end && nums[start] <= nums[pivot]) start++;
        while(start <= end && nums[end]   >= nums[pivot])   end--;
        if(start > end) break;
        swap(nums[start++], nums[end--]);
    }
    swap(nums[--start], nums[pivot]);
    qsort(nums, l, start);
    qsort(nums, start + 1, r);
}

# 最小的K个数

设计一个算法，找出数组中最小的k个数。以任意顺序返回这k个数均可。

#include<algorithm>
#include<vector>
using namespace std;
class Solution {
public:
    int qsort(vector<int>& nums, int start, int end){
        int pivot = start;
        while(true){
            while(start <= end && nums[start] <= nums[pivot]) start++;
            while(start <= end && nums[end]   >= nums[pivot])   end--;
            if(start > end) break;
            swap(nums[start++], nums[end--]);
        }
        swap(nums[--start], nums[pivot]);
        return start;
    }
    vector<int> smallestK(vector<int>& arr, int k) {
        if(k > arr.size()){
             return arr;
        }else if(k <= 0 ){
            return {};
        }else{
            int left = 0, right = arr.size() - 1;
            do{
                int pivot = qsort(arr, left, right);
                if(pivot == k) break;
                else if(pivot > k){
                    // 说明选择的基准太大了，应该在左端区间找
                    right = pivot - 1;
                }else{
                    // 说明选择的基准太小了，应该在右端区间找
                    left = pivot + 1;
                }
            }while(true);
        }
        return vector<int>{arr.begin(), arr.begin()+k};
    }
};
int main(void){
  Solution s;
  vector<int>arr = {1, 7, 2, 6, 3, 4, 5, 0, 9};
  vector<int>nums = s.smallestK(arr, 3);
}

# Merge Sqrt

  • 递归地将元素分成两半，然后将元素按照顺序排列
void merge(vector<int>& nums, int l_start, int l_end, int r_start, int r_end){
  vector<int> tmp;
  int start = l_start;
  while(l_start <= l_end && r_start <= r_end){
    if(nums[l_start] <= nums[r_start]){
      tmp.push_back(nums[l_start++]);
    }else{
      tmp.push_back(nums[r_start++]);
    }
  }

  if(l_start <= l_end){
    while(l_start <= l_end){
      tmp.push_back(nums[l_start ++]);
    }
  }else{
    while(r_start <= r_end){
      tmp.push_back(nums[r_start++]);
    }
  }
  for(auto &c:tmp){
    nums[start++] = c;
  }
}
void merge_sort(vector<int>& nums, int l, int r){
  if (l == r){
    return ;
  }
  /* divide */
  int mid = l + (r - l) / 2;
  merge_sort(nums, l, mid);
  merge_sort(nums, mid + 1, r);

  /*merge*/
  merge(nums, l, mid, mid+1, r);
}

# Select Sqrt

  • 首先，找到最小元素，然后将它和数组的第一个元素交换
  • 然后再在剩下的元素中找到最小元素，将它与数组中第二个位置进行交换。如此往复
#include <iostream>

void print(_RandomAccessIter begin, _RandomAccessIter end){
  for(; begin != end; begin++){
    cout<<*begin<<" ";
  }
  cout<<endl;
}

bool descending(int a, int b){
  return a < b;
}

bool ascending(int a, int b){
  return a > b;
}

template <class _RandomAccessIter , class _Compare>
void select_sort(_RandomAccessIter __first, _RandomAccessIter __last, _Compare cmp){ 
  if (__first == __last){
    return;
  }
  for(; __first != (__last - 1); __first++){
    int idx = 0;
    for(int tmp = 1; (__first + tmp) != __last; tmp++){
      if(cmp(*(__first + idx), *(__first + tmp))){
        idx = tmp;
      }
    }
    iter_swap(__first, (__first + idx));
  }
}
int main(int argc, char* argv[]){
  vector<int>nums1 = {1, 7, 2, 6, 3, 4, 5, 8,  9, 0};
  vector<char>nums2 = {'b', 'e', 'c', 'd', 'f', 'i', 'g', 'z',  'a', 'y'};
  select_sort(nums1.begin(), nums1.end(), descending);
  select_sort(nums2.begin(), nums2.end(), ascending);
  print(nums1.begin(), nums1.end());
  print(nums2.begin(), nums2.end());
  
  return 0;
}

# Insertion sort

​ 插入排序是指在待排序的元素中，假设前面n-1(其中n>=2)个数已经是排好顺序的，现将第n个数插到前面已经排好的序列中，然后找到合适自己的位置，使得插入第n个数的这个序列也是排好顺序的。按照此法对所有元素进行插入，直到整个序列排为有序的过程，称为插入排序。

#include<iostream>
#include<vector>
template<typename T>
void swap(std::vector<T>& nums, int a, int b){
  T tmp;
  tmp = nums[a];
  nums[a] = nums[b];
  nums[b] = tmp; 
}
template<typename T>
void insert_sort(std::vector<T>& nums, int start, int end){
  for(int i = start; i < end; i ++){
    for(int j = i + 1; j > start; j --){
      if(nums[j] < nums[j - 1]){
        swap(nums, j , j - 1);
      }
    }
  }
}
template<typename T>
void print(std::vector<T>nums){

  for(const auto& c: nums){
    std::cout<<c<<" ";
  }
  std::cout<<"\n";
}

int main(int argc, char* argv[]){
  std::vector<char> a = {'r', 'a', 'h', 'w', 'z', 'f'};
  insert_sort(a, 0, a.size() - 1);
  print(a);
}
